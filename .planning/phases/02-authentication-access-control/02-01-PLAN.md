```typescript
if (!accessCheck.authorized) {
  const status = token ? 403 : 401;
  const acceptsHtml = request.headers.get("Accept")?.includes("text/html");

  if (acceptsHtml) {
    // Per D-01: server-side gate returns HTML, not plan content
    const loginUrl = "/api/auth/github/login";
    const returnTo = url.toString();

    if (status === 403) {
      // Authenticated but not on whitelist
      return new Response(accessDeniedHtml(), {
        status: 403,
        headers: { ...cors, "Content-Type": "text/html; charset=utf-8" },
      });
    }

    // Check if token was present but invalid (session expired per D-11)
    if (token) {
      return new Response(sessionExpiredHtml(loginUrl, returnTo), {
        status: 401,
        headers: { ...cors, "Content-Type": "text/html; charset=utf-8" },
      });
    }

    // No token at all — auth required
    return new Response(authRequiredHtml(loginUrl, returnTo), {
      status: 401,
      headers: { ...cors, "Content-Type": "text/html; charset=utf-8" },
    });
  }

  // API clients get JSON (unchanged behavior)
  return Response.json(
    { error: accessCheck.reason || "Access denied" },
    { status, headers: cors }
  );
}
```

Note: The status logic needs refinement. Currently `token ? 403 : 401` is correct when token is valid but user not on ACL. But when token is present and INVALID, we need 401 with session-expired. Check the token validation result: if `token` exists but `user` is undefined (validation failed), that is session expired (D-11). If `user` exists but `checkAccess` failed, that is access denied (403). Adjust the conditional logic to distinguish these three cases:
- No token -> 401 authRequiredHtml
- Token present, validation failed (user is undefined) -> 401 sessionExpiredHtml

```typescript
describe("checkAccess with teams", () => {
  test("returns authorized:true when user is member of a whitelisted team", async () => {
    // Mock fetch for team membership check
    const originalFetch = globalThis.fetch;
    globalThis.fetch = async (url: string | Request | URL) => {
      const urlStr = url.toString();
      if (urlStr.includes("/orgs/myorg/teams/reviewers/memberships/alice")) {
        return new Response(JSON.stringify({ state: "active" }), { status: 200 });
      }
      return new Response("Not found", { status: 404 });
    };

    try {
      const result = await checkAccess(
        { type: "whitelist", teams: ["myorg/reviewers"] },
        { login: "alice", avatar_url: "https://example.com/a.png" },
        "fake-token"
      );
      expect(result.authorized).toBe(true);
    } finally {
      globalThis.fetch = originalFetch;
    }
  });

  test("returns authorized:false when user is NOT member of any whitelisted team", async () => {
    const originalFetch = globalThis.fetch;
    globalThis.fetch = async () => new Response("Not found", { status: 404 });

      const result = await checkAccess(

        "fake-token"

    } finally {
      globalThis.fetch = originalFetch;
    }
  });
});

describe("query param token through auth gate (D-03)", () => {
  test("extractToken reads ?token= query param from request URL", () => {
    // D-03: Share URLs support embedded tokens (?token=xyz) for programmatic access.
    // extractToken() already checks query params — this test confirms the behavior
    // so that the auth gate (which calls extractToken) properly accepts query param tokens.
    const req = new Request("https://share.plannotator.ai/abc123?token=ghp_validtoken123");
    const token = extractToken(req);
    expect(token).toBe("ghp_validtoken123");
  });

  test("valid query param token passes through auth gate (not blocked by HTML 401)", async () => {
    // D-03: When ?token=xyz is valid, the paste content should be served (200),
    // NOT the HTML 401 auth-required error page.
    // This tests the composition: extractToken reads ?token -> validateGitHubToken validates it ->
    // checkAccess authorizes -> paste content returned.
    const originalFetch = globalThis.fetch;
    globalThis.fetch = async (url: string | Request | URL) => {
      const urlStr = url.toString();
      // Mock GitHub /user endpoint to validate the token
      if (urlStr.includes("api.github.com/user")) {
        return new Response(
          JSON.stringify({ login: "ci-bot", avatar_url: "https://example.com/bot.png" }),
          { status: 200, headers: { "Content-Type": "application/json" } }
        );
      }
      return new Response("Not found", { status: 404 });
    };

    try {
      // Step 1: extractToken reads from query param
      const req = new Request("https://share.plannotator.ai/abc123?token=ghp_validtoken123");
      const token = extractToken(req);
      expect(token).toBe("ghp_validtoken123");

      // Step 2: validateGitHubToken confirms the token is valid via GitHub API
      const authResult = await validateGitHubToken(token!);
      expect(authResult.valid).toBe(true);
      expect(authResult.user?.login).toBe("ci-bot");

      // Step 3: checkAccess allows the validated user who is on the whitelist
      const accessResult = await checkAccess(
        { type: "whitelist", users: ["ci-bot"] },
        authResult.user!,
        token
      );
      expect(accessResult.authorized).toBe(true);
      // This full chain proves ?token=xyz bypasses the HTML auth gate when valid
    } finally {
      globalThis.fetch = originalFetch;
    }
  });
});

describe("single permission level (D-16)", () => {
  test("checkAccess returns { authorized: true } with no permission level fields", async () => {
    // D-16: Single permission level — whitelist = full access.
    // checkAccess must return { authorized: true } without readOnly, permissions,
    // accessLevel, or any other granular access fields.
    const originalFetch = globalThis.fetch;
    globalThis.fetch = async () => new Response("Not found", { status: 404 });

    try {
      const result = await checkAccess(
        { type: "whitelist", users: ["alice"] },
        { login: "alice", avatar_url: "https://example.com/a.png" },
        "fake-token"
      );

      // Must be authorized
      expect(result.authorized).toBe(true);

      // D-16: No permission level distinction — only { authorized, reason? } fields
      // Verify no additional permission-related fields exist
      const keys = Object.keys(result);
      expect(keys).not.toContain("readOnly");
      expect(keys).not.toContain("permissions");
      expect(keys).not.toContain("accessLevel");
      expect(keys).not.toContain("role");
      expect(keys).not.toContain("scope");

      // The result shape should only contain authorized (and optionally reason)
      for (const key of keys) {
        expect(["authorized", "reason"]).toContain(key);
      }
    } finally {
      globalThis.fetch = originalFetch;
    }
  });
});
```
