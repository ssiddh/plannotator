2. **Add token validation on mount** (D-09): Add a `useEffect` that validates the token when the component mounts:
   ```typescript
   import React, { createContext, useState, useEffect, useMemo } from "react";
   
   // Inside GitHubProvider:
   useEffect(() => {
     if (!token) return;
     
     let cancelled = false;
     
     (async () => {
       try {
         const res = await fetch("/api/auth/token/validate", {
           method: "POST",
           headers: { "Content-Type": "application/json" },
           body: JSON.stringify({ token }),
         });
         
         if (!res.ok || cancelled) {
           // Token invalid or expired (D-11)
           if (!cancelled) {
             localStorage.removeItem("plannotator_github_token");
             setToken(null);
             setUser(null);
           }
           return;
         }
         
         const data = await res.json() as { valid: boolean; user?: GitHubUser };
         if (!cancelled && data.valid && data.user) {
           setUser(data.user);
         } else if (!cancelled) {
           // Token invalid
           localStorage.removeItem("plannotator_github_token");
           setToken(null);
           setUser(null);
         }
       } catch {
         // Network error -- don't clear token (might be offline)
         // Per D-12: no proactive refresh, reactive handling only
       }
     })();
     
     return () => { cancelled = true; };
   }, [token]);
   ```

3. **Keep all existing stub actions** (syncFromGitHub, syncToGitHub, createPR) -- they are implemented in later phases.

4. **Keep the existing useMemo** for the context value, but ensure it includes `setToken` and `setUser` if needed for future phases. For now the existing memo dependencies `[token, user, prMetadata]` are correct.
