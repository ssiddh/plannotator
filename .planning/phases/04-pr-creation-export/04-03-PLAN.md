    ```typescript
    import { useGitHub } from "@plannotator/github/client/useGitHub";
    import { useGitHubExport } from "@plannotator/github/client/useGitHubExport";
    import { generatePlanHash } from "@plannotator/github/shared/planHash";
    import type { ToastPayload } from "../utils/callback";
    ```

    Hook signature:
    ```typescript
    interface UseGitHubPRExportParams {
      pasteId: string | null;
      markdown: string;
      annotations: Array<{ id: string; blockId: string; type: string; text?: string; originalText: string; images?: Array<{ path: string; name: string }> }>;
      blocks: Array<{ id: string; startLine: number }>;
      setToast: (toast: ToastPayload) => void;
      pasteApiUrl?: string;
    }

    export function useGitHubPRExport({ pasteId, markdown, annotations, blocks, setToast, pasteApiUrl }: UseGitHubPRExportParams) {
    ```

    Internal logic:
    1. **Call useGitHub()** for `isAuthenticated`, `prMetadata`, `setPrMetadata`
    2. **Call useGitHubExport()** for `exportToPR`, `isExporting`, `error`, `lastResult`, `retryAttempt`
    3. **Drift detection** (useEffect):
       - If `prMetadata` has `planHash` and `markdown` is truthy, compute `generatePlanHash(markdown)` and compare
       - Store result in `hasDrift` state
       - Cleanup with cancelled flag
    4. **hasImageAnnotations** (useMemo):
       - `annotations.some(a => a.images && a.images.length > 0)`
    5. **handleExportToPR** (useCallback):
       - If `!pasteId || !markdown` return early
       - Call `exportToPR(pasteId, markdown, annotations, blocks)`
       - On success: call `setPrMetadata(result)`, show success toast with `action: { label: 'View PR', onClick: () => window.open(result.pr_url, '_blank') }`, auto-dismiss after 6s
       - On failure (exportError truthy): show error toast with `action: { label: 'Retry', onClick: () => handleExportToPR() }`, auto-dismiss after 8s
    6. **githubLoginUrl** (useMemo):
       - `pasteApiUrl ? pasteApiUrl + "/api/auth/github/login" : undefined`

    Return all values that ExportModal and App.tsx need:
    ```typescript
    return {
      // For ExportModal props (spread directly)
      isGitHubAuthenticated: isAuthenticated,
      prMetadata,
      onExportToPR: handleExportToPR,
      isExporting,
      retryAttempt,
      exportError: error,
      hasDrift,
      hasImageAnnotations,
      githubLoginUrl,
    };
    ```

    This hook consolidates ALL GitHub PR state and logic into one place. App.tsx calls this hook and spreads the return value into ExportModal props.
