         ```typescript
         for (const summary of summaryAnnotations) {
           const parentGhId = await getGithubCommentId(kv, pasteId, summary.summarizesThreadId);
           if (!parentGhId) {
             result.errors.push(`Summary for thread ${summary.summarizesThreadId}: parent not synced to GitHub`);
             continue;
           }
           // Post summary as reply
           const reply = await replyToComment(owner, repo, prNumber, parentGhId, summary.text || summary.originalText, token);
           await setMapping(kv, pasteId, summary.id, reply.id);
           result.synced++;

           // Attempt thread resolution (D-09)
           const threadMap = await fetchReviewThreads(owner, repo, prNumber, token);
           const threadInfo = threadMap.get(parentGhId);
           if (threadInfo) {
             const resolved = await resolveReviewThread(threadInfo.threadNodeId, token);
             if (!resolved) {
               // D-11/D-34: graceful failure, summary still posted
               result.warnings = result.warnings || [];
               result.warnings.push(`Summary posted but thread not resolved -- resolve manually on GitHub`);
             }
           }
         }
         ```
