**2. Initialize the sync hook:**

Find where `useGitHubPRExport` is initialized (search for `useGitHubPRExport` in the file). Add the sync hook near it:

```typescript
// GitHub inbound sync
const {
  annotations: syncAnnotations,
  isLoading: isSyncLoading,
  isSyncing,
  newCommentCount,
  syncFromGitHub,
  lastSync: lastSyncTime,
  error: syncError,
} = useGitHubPRSync({
  pasteId: pasteId || "",
  prMetadata: github?.prMetadata || null,
  blocks,
  token: github?.token || null,
  pasteServiceUrl: pasteServiceUrl || undefined,
  pollInterval: 300000, // 5 minutes per D-05
  enabled: !!github?.prMetadata && !!github?.token,
  onSyncComplete: (stats) => {
    if (stats.new > 0) {
      setToast({
        type: "success",
        message: `Synced ${stats.new} comment${stats.new === 1 ? "" : "s"} from GitHub`,
      });
    } else if (stats.updated > 0 || stats.deleted > 0) {
      setToast({
        type: "success",
        message: `Updated ${stats.updated} and removed ${stats.deleted} comment${stats.deleted === 1 ? "" : "s"}`,
      });
    } else {
      setToast({
        type: "info",
        message: "Already up to date",
      });
    }
  },
  onError: (errorMsg, errorType) => {
    if (errorType === "token_expired") {
      setToast({
        type: "error",
        message: "Session expired. Please sign in again.",
      });
      // D-15: redirect to OAuth
      window.location.href = `/api/auth/github/login?return_to=${encodeURIComponent(window.location.href)}`;
    } else if (errorType === "rate_limit") {
      const resetAt = errorMsg.split(":")[1];
      const resetTime = resetAt
        ? new Intl.DateTimeFormat(undefined, { hour: "numeric", minute: "2-digit" }).format(new Date(parseInt(resetAt) * 1000))
        : "soon";
      setToast({
        type: "warning",
        message: `Rate limit hit. Retrying at ${resetTime}`,
      });
    } else {
      setToast({
        type: "error",
        message: "Sync failed. Check your connection.",
        action: {
          label: "Retry",
          onClick: () => syncFromGitHub(),
        },
      });
    }
  },
});
```

Note: `pasteId`, `github` (from useGitHub context), `blocks`, `setToast`, and `pasteServiceUrl` should already be available in the component scope. Find the actual variable names by reading the file:
- `pasteId` might be from URL params or server response
- `github` context is likely from `useGitHub()` hook
- `blocks` from `parseMarkdownToBlocks`
- `setToast` from existing toast state

**4. Merge sync annotations with local annotations:**

Find where annotations are rendered (passed to `AnnotationPanel` and `Viewer`). The app likely has a local `annotations` state array. Merge sync annotations:

```typescript
const allAnnotations = useMemo(() => {
  // Combine local annotations with synced GitHub annotations
  // Synced annotations have source: "github-pr", local ones don't
  const localAnns = annotations.filter(a => a.source !== "github-pr");
  return [...localAnns, ...syncAnnotations];
}, [annotations, syncAnnotations]);
```

Use `allAnnotations` wherever `annotations` is currently passed to `AnnotationPanel` and `Viewer`. Search for `annotations={annotations}` in the JSX and replace with `annotations={allAnnotations}`.
